281 if (it->time < temp->time) { sum += it->time; }

314float avg_RR_priorita (Processo_log *arr, int num_processi, int size) {
315 float sum = 0;
316
318
319 map<string, int> map_max;
320
327 map<string, int> map_sum;
328
330
331 for (int i = 0; i < size; i++) {
332 for (int j = 0; j < size; j++) {
333 if (arr[i].nome == arr[j].nome && arr[i].time <= arr[j].time) {
334 map_max[arr[j].nome] = arr[j].time;
335 }
336 }
337 }
338
339 for (int i = 0; i < size; i++) {
340 if (arr[i].time < map_max[arr[i].nome]) {
341 map_sum[arr[i].nome] += arr[i+1].time - arr[i].time - 1;
342 }
343 }
344
345 for (int i = 0; i < num_processi; i++) {
346 sum += map_max[arr[i].nome] - map_sum[arr[i].nome];
347 }
348
349 float size_dec = static_cast<float>(num_processi);
350 return sum/size_dec;
